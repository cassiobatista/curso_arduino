\section{Introdução}
\subsection{Introdução}
\begin{frame}{Introdução}
\begin{itemize}
	\item Baseado em ``\textit{Introduction to Arduino}'', de Alan G. Smith
\end{itemize}
\begin{block}{O que é Arduino?}
\begin{itemize}
	\item O \textit{Hardware};
	\item O ambiente de desenvolvimento;
	\item A comunidade (\url{https://www.arduino.cc})
\end{itemize}
\end{block}
\begin{itemize}
	\item Arduino é a placa de desenvolvimento construída sobre o 
	microcontrolador ATmega, da fabricante Atmel, contendo uma IDE para 
	programação e um fórum/comunidade para discussão. \medskip
	\item \textit{Open-source prototyping platform based on easy-to-use hardware
	and software}%\footnote{https://www.arduino.cc/en/Guide/Introduction}
\end{itemize}
\end{frame}

\begin{frame}{Vantagens}
\begin{itemize}
	\item Barato: U\$ 50.00, em média \medskip
	\item \textit{Cross-Platform}: Windows, OSX, Linux. \medskip
	\item Ambiente de programação: IDE \textit{easy-to-use} para iniciantes \medskip
	\item \textit{Open-source software}: Pode ser expandido através do C++ \medskip
	\item \textit{Open-source hardware}: Designers de circuito podem fazer sua própria versão \medskip
\end{itemize}
\end{frame}

\begin{frame}{Arduino UNO --- Especificações}
\begin{itemize}
	\item Microcontrolador: ATmega328P \medskip
	\item 14 pinos digitais de I/O (6 de PWM) \medskip
	\item 6 pinos analógicos \medskip
	\item 32~KB de memória Flash (armaz. \textit{sketch}) \medskip
	\item 2~KB de memória SRAM (criação e manipulação de variáveis) \medskip
	\item 1~KB de memória EEPROM \medskip
	\item 16~MHz de Clock
\end{itemize}
\end{frame}

\begin{frame}{MicroControlador x MicroProcessador}
\begin{itemize}
	\item Ambos realizam operações de busca, decodificação e execução; \medskip
	\item Microprocessador: CI responsável pelo processamento de dados, como uma
	ULA, mas precisa receber ordens externas e ter outros componentes externos 
	para funcionar; \medskip
	\item Microcontrolador: Contém $\mu$processador, RAM, ROM, \textit{timers}, 
	I/Os... tudo em um único CI;
\end{itemize}
\begin{block}{O Arduino é o quê, afinal?}
\begin{itemize}
	\item Arduino é a PCB criada para funcionar sobre o microcontrolador ATmega,
	da Atmel.
\end{itemize}
\end{block}
\end{frame}
