\begin{frame}{\textit{Overview}}
\begin{itemize}
	\item Baseado em ``\textit{Introduction to Arduino}'', de Alan G. Smith
\end{itemize}
\begin{block}{O que é Arduino?}
\begin{itemize}
	\item O \textit{Hardware};
	\item O ambiente de desenvolvimento;
	\item A comunidade (\url{https://www.arduino.cc})
\end{itemize}
\end{block}
\begin{itemize}
	\item Arduino é a placa de desenvolvimento construída sobre o 
	microcontrolador ATmega, da fabricante Atmel, contendo uma IDE para 
	programação e um fórum/comunidade para discussão. \medskip
	\item \textit{Open-source prototyping platform based on easy-to-use hardware
	and software}%\footnote{https://www.arduino.cc/en/Guide/Introduction}
\end{itemize}
\end{frame}

\begin{frame}{Vantagens}
\begin{itemize}
	\item Barato: U\$ 50.00, em média \medskip
	\item \textit{Cross-Platform}: Windows, OSX, Linux. \medskip
	\item Ambiente de programação: IDE \textit{easy-to-use} para iniciantes \medskip
	\item \textit{Open-source software}: Pode ser expandido através do C++ \medskip
	\item \textit{Open-source hardware}: Designers de circuito podem fazer sua própria versão \medskip
\end{itemize}
\end{frame}

\begin{frame}{Arduino UNO --- Especificações}
\begin{itemize}
	\item Microcontrolador: ATmega328P \medskip
	\item 14 pinos digitais de I/O (6 de PWM) \medskip
	\item 6 pinos analógicos \medskip
	\item 32~KB de memória Flash (armazenamento de \textit{sketch}) \medskip
	\item 2~KB de memória SRAM (criação e manipulação de variáveis) \medskip
	\item 1~KB de memória EEPROM \medskip
	\item 16~MHz de Clock
\end{itemize}
\end{frame}

\begin{frame}{MicroControlador $\times$ MicroProcessador}
\begin{itemize}
	\item Ambos realizam operações de busca, decodificação e execução; \medskip
	\item Microprocessador: CI responsável pelo processamento de dados, como uma
	ULA, mas precisa receber ordens externas e ter outros componentes externos 
	para funcionar; \medskip
	\item Microcontrolador: Contém $\mu$processador, RAM, ROM, \textit{timers}, 
	I/Os... tudo em um único CI;
\end{itemize}
\begin{block}{O Arduino é o quê, afinal?}
\begin{itemize}
	\item Arduino é a PCB criada para funcionar sobre o microcontrolador ATmega,
	da Atmel.
\end{itemize}
\end{block}
\end{frame}

\begin{frame}{Pinos de Alimentação}
\begin{itemize}
	\item \textbf{V$_{\text{in}}$}: Entrada de alimentação da placa quando 
	utilizada uma fonte externa. \medskip
	\item \textbf{5V}: Fonte de alimentação para o microcontrolador e outros
	componentes da placa. \medskip
	\item \textbf{3V3}: Alimentação de 3,3 V fornecida pelo controlador USB. \medskip
	\item \textbf{GND} (\textit{ground}): Pino terra. \medskip
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Programação}
\begin{itemize}
	\item A linguagem utilizada no Arduino tem como referência C++. \smallskip
	\item \texttt{setup()}: função utilizada para inicializar valores assim que a placa é ligada. \smallskip
	\item \texttt{loop()}: função executada várias vezes até que haja alguma interferência. \smallskip
	\item \texttt{pinMode()}: Inicializa um pino como entrada (INPUT) ou saída 
	(OUTPUT).
	\begin{lstlisting}
	pinMode(2, INPUT); //Setando o pino 2 como entrada
	pinMode(13, OUTPUT); //Setando o pino 13 como saída
	\end{lstlisting}
\end{itemize}
\end{frame}

\begin{frame}[fragile]{Programação}
\begin{itemize}
	\item \texttt{digitalWrite()}: Escreve uma informação digital, ou seja, 0 
	(LOW, 0V) ou 1 (HIGH, 5V) em um pino de saída. \smallskip
	\begin{lstlisting}
	digitalWrite(13, HIGH); //tensão alta no pino 13 
	digitalWrite(13, LOW); //tensão baixa no pino 13
	\end{lstlisting}

	\item \texttt{digitalRead()}: Lê uma informação digital, ou seja, 0 
	(0V) ou 1 (5V) em um pino de entrada. \smallskip
	\begin{lstlisting}
	int valor = digitalRead(2); //Lê pino 2
	\end{lstlisting}

	\item \texttt{analogWrite()}: Escreve uma informação analógica, ou seja, um 
	valor inteiro de 8 bits de 0 à 255 em um pino de saída \medskip
	\begin{lstlisting}
	analogWrite(13, 250); //tensão [0V--5V] [0--255] pino 13 
	\end{lstlisting}

	\item \texttt{analogRead()}: Lê uma informação analógica, geralmente de 
	sensores
	\begin{lstlisting}
	int valor = analogRead(A0); //Lê pino analógico 0
	\end{lstlisting}
\end{itemize}
\end{frame}
